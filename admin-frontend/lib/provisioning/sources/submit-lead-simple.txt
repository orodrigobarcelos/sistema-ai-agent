import { createClient } from "https://esm.sh/@supabase/supabase-js@2";

const corsHeaders = {
  "Access-Control-Allow-Origin": "*",
  "Access-Control-Allow-Headers": "authorization, x-client-info, apikey, content-type",
};

interface CreateLeadPayload {
  name: string;
  whatsapp: string;
  email?: string | null;
  country_code?: string;
  utm_source?: string | null;
  utm_campaign?: string | null;
  utm_medium?: string | null;
  utm_content?: string | null;
  utm_term?: string | null;
}

Deno.serve(async (req) => {
  if (req.method === "OPTIONS") {
    return new Response("ok", { headers: corsHeaders });
  }

  try {
    const supabase = createClient(
      Deno.env.get("SUPABASE_URL")!,
      Deno.env.get("SUPABASE_SERVICE_ROLE_KEY")!
    );

    const payload = await req.json() as CreateLeadPayload;
    const { name, whatsapp, email, country_code, utm_source, utm_campaign, utm_medium, utm_content, utm_term } = payload;

    console.log(`[submit-lead-simple] Recebido: ${name} - ${whatsapp}`);

    if (!name?.trim() || !whatsapp?.trim()) {
      return new Response(
        JSON.stringify({ error: "Nome e WhatsApp sao obrigatorios" }),
        { status: 400, headers: { ...corsHeaders, "Content-Type": "application/json" } }
      );
    }

    const { data: existingLead } = await supabase
      .from("leads")
      .select("id")
      .eq("whatsapp", whatsapp)
      .maybeSingle();

    if (existingLead) {
      console.log(`[submit-lead-simple] Lead ja existe: ${existingLead.id}`);
      const supabaseUrl = Deno.env.get("SUPABASE_URL")!;
      fetch(`${supabaseUrl}/functions/v1/process-lead-signup-simple`, {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({ lead_id: existingLead.id }),
      }).catch((err) => console.error("[submit-lead-simple] error:", err));

      return new Response(
        JSON.stringify({ success: true, lead_id: existingLead.id }),
        { headers: { ...corsHeaders, "Content-Type": "application/json" } }
      );
    }

    const digits = whatsapp.replace(/\D/g, "");
    const sessionId = `${digits}@s.whatsapp.net`;

    const { data: newLead, error: insertError } = await supabase
      .from("leads")
      .insert({
        name: name.trim(),
        whatsapp,
        email: email?.trim() || null,
        country_code: country_code || "+55",
        session_id: sessionId,
        utm_source, utm_campaign, utm_medium, utm_content, utm_term,
      })
      .select("id")
      .single();

    if (insertError) {
      console.error("[submit-lead-simple] Insert error:", insertError);
      return new Response(
        JSON.stringify({ error: "Erro ao criar lead" }),
        { status: 500, headers: { ...corsHeaders, "Content-Type": "application/json" } }
      );
    }

    console.log(`[submit-lead-simple] Lead criado: ${newLead.id}`);
    const supabaseUrl = Deno.env.get("SUPABASE_URL")!;
    fetch(`${supabaseUrl}/functions/v1/process-lead-signup-simple`, {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({ lead_id: newLead.id }),
    }).catch((err) => console.error("[submit-lead-simple] error:", err));

    return new Response(
      JSON.stringify({ success: true, lead_id: newLead.id }),
      { headers: { ...corsHeaders, "Content-Type": "application/json" } }
    );

  } catch (error) {
    console.error("[submit-lead-simple] Error:", error);
    return new Response(
      JSON.stringify({ error: "Erro interno do servidor" }),
      { status: 500, headers: { ...corsHeaders, "Content-Type": "application/json" } }
    );
  }
});
